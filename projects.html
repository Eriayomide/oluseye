def check_permission(user: User, resource: str, action: str) -> bool:
    # Resolve role hierarchy: church → region → country → HQ
    role_chain = get_role_chain(user.role, user.scope_level)

    for role in role_chain:
        policy = PERMISSION_MATRIX.get(role, {})
        if action in policy.get(resource, []):
            return True

    raise PermissionDenied(
        f"Role '{user.role}' cannot '{action}' on '{resource}'"
    )

3-Level Distress Detector — Pidgin-aware 
CRISIS_SIGNALS = [
    "i want to end it all", "no reason to continue",
    "i wan just disappear", "make god just take me",
    "i don plan how i go do am",
]

AMBER_SIGNALS = [
    "i tire for this life", "wetin be the point",
    "nobody care if i dey here", "i don give up",
]

def classify_distress(message: str) -> DistressLevel:
    text = message.lower().strip()
    if any(sig in text for sig in CRISIS_SIGNALS):
        return DistressLevel.RED    # immediate crisis
    if any(sig in text for sig in AMBER_SIGNALS):
        return DistressLevel.AMBER  # check in deeper
    return DistressLevel.GREEN   # standard conversation

def retrieve_and_respond(query: str, index: faiss.Index) -> str:
    # Embed the user query
    query_vec = embedder.encode([query])

    # Retrieve top-3 relevant document chunks
    _, indices = index.search(query_vec, k=3)
    context = "\n\n".join(doc_store[i] for i in indices[0])

    # Assemble grounded prompt for Claude
    prompt = (
        f"Answer using only the context below.\n\n"
        f"Context:\n{context}\n\n"
        f"Question: {query}"
    )
    return call_claude(prompt)

def decode_vin(vin: str) -> VehicleRecord:
    if len(vin) != 17:
        raise ValueError("VIN must be exactly 17 characters")

    record = VehicleRecord(
        country   = VIN_COUNTRY_MAP.get(vin[:2], "Unknown"),
        make      = WMI_MAP.get(vin[:3], "Unknown"),
        model_year = YEAR_MAP.get(vin[9], "Unknown"),
        serial    = vin[11:],
    )

    # Flag suspicious patterns via Claude API
    record.anomaly = check_anomaly(vin, record)
    return record
